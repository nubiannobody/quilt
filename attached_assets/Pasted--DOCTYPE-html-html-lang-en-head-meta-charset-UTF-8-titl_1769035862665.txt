        const business = names[i];

        const affinities = [];

          id: i + 1,
          name: business.name,
          logo: business.logo,
          tagline: business.tagline,
          website: business.website,
          industry,
          affinities
        });

                    </div>
                  )}
                </div>

                {business.website && (
                  <a
                    href={business.website}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="flex items-center gap-1 text-xs text-blue-600 hover:text-blue-800 font-medium mt-auto"
                    onClick={(e) => e.stopPropagation()}
                  >
                    <Globe size={12} />
                    Visit Website
                  </a>
                )}
              </div>
            </div>
          </div>
        </div>
      );
    };

    const BusinessQuilt = () => {
      const [businesses] = useState(generateSampleBusinesses());
      const [selectedIndustry, setSelectedIndustry] = useState(null);
      const [selectedAffinities, setSelectedAffinities] = useState([]);

      const filteredBusinesses = useMemo(() => {
        return businesses.filter(business => {
          const industryMatch = !selectedIndustry || business.industry === selectedIndustry;
          const affinityMatch = selectedAffinities.length === 0 ||
            selectedAffinities.every(aff => business.affinities.includes(aff));
          return industryMatch && affinityMatch;
        });
      }, [businesses, selectedIndustry, selectedAffinities]);

      const toggleAffinity = (aff) => {
        setSelectedAffinities(prev =>
          prev.includes(aff) ? prev.filter(a => a !== aff) : [...prev, aff]
        );
      };

      return (
        <div className="min-h-screen bg-gradient-to-br from-gray-50 to-gray-100 p-4 sm:p-6">
          <div className="max-w-7xl mx-auto">
            <div className="text-center mb-6">
              <h1 className="text-4xl font-bold text-gray-800 mb-2">Business Quilt</h1>
              <p className="text-gray-600">Discover local businesses in our community — click to flip!</p>
            </div>

            {/* Filters */}
            <div className="bg-[#2F4F3E] rounded-lg shadow-md p-4 mb-6">
              <div className="flex flex-wrap gap-3 items-center">
                <select
                  value={selectedIndustry || ''}
                  onChange={(e) => setSelectedIndustry(e.target.value || null)}
                  className="border rounded px-3 py-2 text-sm bg-white"
                >
                  <option value="">All Industries</option>
                  {Object.keys(INDUSTRIES).map(ind => (
                    <option key={ind} value={ind}>{ind}</option>
                  ))}
                </select>

                <select
                  value=""
                  onChange={(e) => {
                    if (e.target.value) { toggleAffinity(e.target.value); e.target.value = ''; }
                  }}
                  className="border rounded px-3 py-2 text-sm bg-white"
                >
                  <option value="">Filter Affinity</option>
                  {Object.keys(AFFINITIES).map(aff => (
                    <option key={aff} value={aff}>{aff}</option>
                  ))}
                </select>

                {(selectedIndustry || selectedAffinities.length > 0) && (
                  <button
                    onClick={() => { setSelectedIndustry(null); setSelectedAffinities([]); }}
                    className="text-sm text-white hover:text-gray-200 underline"
                  >
                    Clear all filters
                  </button>
                )}
              </div>

              {selectedAffinities.length > 0 && (
                <div className="flex flex-wrap gap-2 mt-3">
                  {selectedAffinities.map(aff => (
                    <span key={aff}
                      className="px-3 py-1 rounded-full text-white text-xs flex items-center gap-2"
                      style={{ backgroundColor: AFFINITIES[aff] }}
                    >
                      {aff}
                      <button onClick={() => toggleAffinity(aff)} className="hover:opacity-80">
                        <X size={14} />
                      </button>
                    </span>
                  ))}
                </div>
              )}
            </div>

            {/* Quilt Grid */}
            <div className="grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 xl:grid-cols-6 gap-4 bg-[#355E3B] p-6 rounded-xl shadow-xl">
              {filteredBusinesses.map(business => (
                <BusinessCard key={business.id} business={business} />
              ))}
            </div>

            {filteredBusinesses.length === 0 && (
              <div className="text-center py-12 text-gray-500">
                No businesses match your filters.
              </div>
            )}
          </div>
        </div>
      );
    };

    const root = ReactDOM.createRoot(document.getElementById('root'));
    root.render(<BusinessQuilt />);
  </script>
